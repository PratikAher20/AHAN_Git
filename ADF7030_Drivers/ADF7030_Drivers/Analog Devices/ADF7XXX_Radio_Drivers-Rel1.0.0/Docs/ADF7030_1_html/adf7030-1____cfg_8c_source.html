113 ADI_ADF7030_1_RESULT adf7030_1__SendConfiguration(
114  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,
115  uint8_t * pCONFIG,
116  uint32_t Size
117 )
118 {
119 
120  uint32_t array_position = 0;
121 
122  do
123  {
124  // Calculate the number of bytes to write
125  uint32_t length = (*(pCONFIG + array_position ) << 16) |
126  (*(pCONFIG + array_position + 1) << 8) |
127  (*(pCONFIG + array_position + 2));
128 
129  if(length > 0xFFFF)
130  {
131  return ADI_ADF7030_1_SPI_COMM_FAILED;
132  }
133 
134  // Write the SPI data pointed to location (MEMORY_FILE + array_position) with specified length (length)
135  uint8_t * pSeqData = (pCONFIG + array_position + 3);
136 
137  // Transfer the Configuration sequence
138  if(adf7030_1__XferCfgSeq( pSPIDevInfo,
139  pSeqData,
140  length - 4) != ADI_ADF7030_1_SUCCESS)
141  {
142  return ADI_ADF7030_1_SPI_COMM_FAILED;
143  }
144 
145  // Update the array position to point to the next block
146  array_position += length;
147 
148  }while(array_position < Size); // Continue operation until full data file has been written
149 
150  return ADI_ADF7030_1_SUCCESS;
151 }

172 ADI_ADF7030_1_RESULT adf7030_1__XferCfgSeq(
173  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,

176 )
177 {
178  // Retrieve Size actual size of sequence
179  switch( *pData & 0x78 )
180  {
181  // 32bits Address Block Write Access of the "Configuration Format"
182  case( ADF703x_SPI_MEM_WRITE | ADF703x_SPI_MEM_BLOCK | ADF703x_SPI_MEM_ADDRESS | ADF703x_SPI_MEM_LONG ) :
183  {
184  // Move cfg sequence pointer by 1 byte
185  pData++;
186 
187  // Get the Xfer byte size
188  uint32_t SeqDataLen = (Size - 4);
189 
190  // Get the PHY Radio where to write sequence + change pointer to start of data sequence
191  uint32_t Addr = __REV(*(uint32_t *)pData);
192 
193  // Currently, the RADIO SPI driver is implemented to provide a one-to-one relationship
194  // between memory bloc representation of the PHY Radio and the HOST. (Provided the underlying endianness is the same)
195  // Byte transfer via the adf7030_1 SPI interface is done in natural fashion (byteaddress increment)
196  // However 32bits word transfers are wordaddress increment. Each word must byte reversed (intrinsic __REV() on ARM Cortex does just that)
197  // Current driver SPIs call are already rev-ing words to comply with SPI word transfer protocol
198 
199  // This version of ADI Radio SPI driver does not fully support the recommended *.cfg file format where
200  // rev-ing is already done as part of the sequence
201 
202  // So for now, we just need to unrev data...
203  uint32_t * pUNREV_BUFF = (uint32_t *)spi_CfgBuffer;
204  uint32_t * pSEQ_BUFF = (uint32_t *)pData;
205 
206  pSEQ_BUFF++;
207 
208  for(uint32_t i = 0; i < (SeqDataLen >> 2); i++)
209  {
210  *pUNREV_BUFF++ = __REV(*pSEQ_BUFF++);
211  }
212 
213  // Transfert data to using the existing RADIO SPI driver
214  if(adf7030_1__SPI_wr_word_b_a( pSPIDevInfo,
215  Addr,
216  SeqDataLen >> 2,
217  (uint32_t *)spi_CfgBuffer) != ADI_ADF7030_1_SUCCESS)
218  {
219  return ADI_ADF7030_1_SPI_COMM_FAILED;
220  }
221  break;
222  }
