114  void *pCBParam,
115  uint32_t Port,
116  void *PinIntData
117 )
118 {
119  /* Pointer to Device Instance data */
120  ADI_ADF7030_1_INFO * pDevInfo = (ADI_ADF7030_1_INFO *) pCBParam;
121 
122  uint32_t irq_event = 0;
123 
124  if((Port == pDevInfo->IntGPIOInfo[ADI_ADF7030_1_INTPIN0].ePort) &&
125  (*(uint32_t *)PinIntData & pDevInfo->IntGPIOInfo[ADI_ADF7030_1_INTPIN0].nPin))
126  {
127  /* If callback Port and pin matches the Instance Irq0 port & pin definition */
128  irq_event = ADI_ADF7030_1_EVENT_INT0;
129  }
130  else if((Port == pDevInfo->IntGPIOInfo[ADI_ADF7030_1_INTPIN1].ePort) &&
131  (*(uint32_t *)PinIntData & pDevInfo->IntGPIOInfo[ADI_ADF7030_1_INTPIN1].nPin))
132  {
133  /* If callback Port and pin matches the Instance Irq1 port & pin definition */
134  irq_event = ADI_ADF7030_1_EVENT_INT1;
135  }
136  else
137  {
138  /* Should not end up here */
139  }
140 
141  /* If global Radio driver application callback is defined */
142  if(pDevInfo->pfCallback != NULL)
143  {
144  /* Execute the global radio driver event callback */
145  pDevInfo->pfCallback(pDevInfo, irq_event, NULL);
146  }
147 
148  return;
