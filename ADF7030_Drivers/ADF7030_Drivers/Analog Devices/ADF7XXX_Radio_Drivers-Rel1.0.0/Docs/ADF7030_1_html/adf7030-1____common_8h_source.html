53 #if defined ( __ICCARM__ )
54 /*
55 * IAR MISRA C 2004 error suppressions.
56 *
57 *
58 * Pm120 (rule 19.10): In the definition of a function-like macro each parameter
59 * shall be enclosed in parenthesis.
60 * This is not possible in attributes and pragmas
61 * Pm154 (rule 19.13): The # and ## preprocessor operators shall not be used.
62 * We need to do this to abstract the macros for the
63 * different toolchains
64 */
65 #pragma diag_suppress=Pm120,Pm154
66 #endif
67 
68 #define PRAGMA(x) _Pragma(#x)
69 #define ATTRIBUTE(x) __attribute__((x))
70 
71 #if defined (__GNUC__)
72  /* Gcc uses attributes */
73  #define ADI_ALIGNED_PRAGMA(num)
74  #define ADI_ALIGNED_ATTRIBUTE(num) ATTRIBUTE(aligned(num))
75  #define ADI_UNUSED_ATTRIBUTE ATTRIBUTE(unused)
76 
77  #define KEEP_VAR(var) var ATTRIBUTE(used)
78 
79 #elif defined ( __ICCARM__ )
80  /* IAR uses a pragma */
81  #define ADI_ALIGNED_ATTRIBUTE(num)
82  #define ADI_ALIGNED_PRAGMA(num) PRAGMA(data_alignment=num)
83  #define ADI_UNUSED_ATTRIBUTE
84 
85  #define KEEP_VAR(var) __root var
86 
87 #elif defined (__CC_ARM)
88  /* Keil uses a decorator which is placed in the same position as pragmas */
89  #define ADI_ALIGNED_ATTRIBUTE(num)
90  #define ADI_ALIGNED_PRAGMA(num) __align(##num)
91  #define ADI_UNUSED_ATTRIBUTE ATTRIBUTE(unused)
