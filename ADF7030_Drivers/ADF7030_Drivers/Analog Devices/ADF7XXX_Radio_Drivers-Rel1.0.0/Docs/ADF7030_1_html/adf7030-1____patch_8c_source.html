217 ADI_ADF7030_1_RESULT adf7030_1__SelfCheckPatch(
218  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,

220  bool bCHECK,
221  ADI_ADF7030_1_INTPIN eIntPin,
222  ADI_ADF7030_1_RADIO_STATE eCmd,

224 )

226  /* ADF7030-1 result code */
227  ADI_ADF7030_1_RESULT eResult;
228 
229  DATA_BLOCK_DESCRIPTOR_Type * pPATCH_SelfcheckBlock = &pPATCH->BLOCKS[pPATCH->CheckBlockId];
230 
231  /* Download the patch selfcheck code */
232  if((eResult = adf7030_1__WritePatchBlock( pSPIDevInfo,
233  pPATCH_SelfcheckBlock)) != ADI_ADF7030_1_SUCCESS)
234  {
235  return eResult;
236  }
237 
238  if(bCHECK == true)
239  {
240  /* Verify the patch selfcheck code integrity */
241  if((eResult = adf7030_1__VerifyPatchBlock( pSPIDevInfo,
242  pPATCH_SelfcheckBlock )) != ADI_ADF7030_1_SUCCESS)
243  {
244  return eResult;
245  }
246  }
247 
248  /* Setup patch selfcheck checksum descriptor */
249  if(pCHECKSUM_CFG == NULL) pCHECKSUM_CFG = pPATCH->pChecksum;
250 
251  /* Configure patch selfcheck */
252  if((eResult = adf7030_1__WritePatchCheckumInfo( pSPIDevInfo,
253  pPATCH,
254  pCHECKSUM_CFG )) != ADI_ADF7030_1_SUCCESS)
255  {
256  return eResult;
257  }
258 
259  /* Enabling the patch selfcheck function */
260  if((eResult = adf7030_1__SetupPatch( pSPIDevInfo,
261  pPATCH->CheckSeq,
262  true )) != ADI_ADF7030_1_SUCCESS)
263  {
264  return eResult;
265  }
266 
267  /* Pointer to the ADF7030-1 instance information */
268  ADI_ADF7030_1_INFO * pDevInfo = (ADI_ADF7030_1_INFO *)pSPIDevInfo->hDevInfo;
269 
270  /* Clearing all events on patch selfcheck IRQ pin */
271  if((eResult = adf7030_1__IRQ_ClrAllStatus( pDevInfo, eIntPin)) != ADI_ADF7030_1_SUCCESS )
272  {
273  return eResult;
274  }
275 
276  /* Save current IRQ Map config */
277  if((eResult = adf7030_1__IRQ_GetMap(pDevInfo, eIntPin)) != ADI_ADF7030_1_SUCCESS )
278  {
279  return eResult;
280  }
281 
282  uint32_t tmp_nIntMap = pDevInfo->IntGPIOInfo[eIntPin].nIntMap;
283 
284  /* Setting BUFF_HALF_IRQn_Msk and BUFF_FULL_IRQn_Msk event for patch selfcheck IRQ pin */
285  if((eResult = adf7030_1__IRQ_SetMap( pDevInfo,
286  eIntPin,
287  BUFF_HALF_IRQn_Msk | BUFF_FULL_IRQn_Msk )) != ADI_ADF7030_1_SUCCESS )
288  {
289  return eResult;
290  }
291 
292  /* Override IRQ callback function */
293  if((eResult = adf7030_1__IRQ_SetCallback( pDevInfo,
294  eIntPin,
295  adf7030_1__SelfCheckPatch_callback )) != ADI_ADF7030_1_SUCCESS )
296  {
297  return eResult;
298  }
299 
300  gbselfcheck_finished = false;
301 
302  /* Disable PHY Error readback if required */
303  bool bPhyErrorCheck = pSPIDevInfo->bPhyErrorCheck;
304  if(bPhyErrorCheck == true) pSPIDevInfo->bPhyErrorCheck = false;
305 
306  /* Trigger the Offline calibration selfcheck execution */
307  if((eResult = adf7030_1__STATE_PhyCMD( pSPIDevInfo, eCmd )) != ADI_ADF7030_1_SUCCESS)
308  {
309  return eResult;
310  }
311 
312  /* Wait for interrupt */
313  while(gbselfcheck_finished == false) __WFE();
314 
315  /* Recall PHY Error readback setting */
316  if(bPhyErrorCheck == true) pSPIDevInfo->bPhyErrorCheck = bPhyErrorCheck;
317 
318  /* Read IRQ status register of selfcheck IRQ pin */
319  if((eResult = adf7030_1__IRQ_GetStatus( pDevInfo, eIntPin )) != ADI_ADF7030_1_SUCCESS )
320  {
321  return eResult;
322  }
323 
324  eResult = (pDevInfo->IntGPIOInfo[eIntPin].nIntStatus & BUFF_FULL_IRQn_Msk) ? ADI_ADF7030_1_SUCCESS : ADI_ADF7030_1_FAILURE;
325 
326  /* Clear selfcheck IRQ pin */
327  if((eResult = adf7030_1__IRQ_ClrStatus( pDevInfo,
328  eIntPin,
329  pDevInfo->IntGPIOInfo[eIntPin].nIntStatus )) != ADI_ADF7030_1_SUCCESS )
330  {
331  return eResult;
332  }
333 
334  /* Restore the default IRQ callback function */
335  if((eResult = adf7030_1__IRQ_SetCallback( pDevInfo,
336  eIntPin,
337  adf7030_1__IRQ_Callback )) != ADI_ADF7030_1_SUCCESS )
338  {
339  return eResult;
340  }
341 
342  /* Restore IRQ Map */
343  eResult = adf7030_1__IRQ_SetMap( pDevInfo,
344  eIntPin,
345  tmp_nIntMap );
346 
347  return eResult;
348 }

408 ADI_ADF7030_1_RESULT adf7030_1__SetupPatch(
409  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,
410  uint16_t SetupSeq,
411  bool bCHECK
412 )
413 {
414  /* ADF7030-1 result code */
415  ADI_ADF7030_1_RESULT eResult;
416 
417  if(SetupSeq == 0)

430  {
431  return eResult;
432  }
433 
434  /* Readback Inject Sequence number if requested */
435  if(bCHECK == true)
436  {
437  uint32_t Seq_rb;
438 
439  if((eResult = adf7030_1__SPI_rd_word_b_a( pSPIDevInfo,
440  0x20000130,
441  1,
442  &Seq_rb )) != ADI_ADF7030_1_SUCCESS)
443  {
444  return eResult;
445  }
446 
447  /* Compare Setup patch sequence with the readback sequence */
448  if(Seq != Seq_rb)
449  {
450  eResult = ADI_ADF7030_1_FAILURE;
451  }
452  }
453 
454  return eResult;

607 ADI_ADF7030_1_RESULT adf7030_1__XferPatchBlock(

609  DATA_BLOCK_DESCRIPTOR_Type * pBLOCK,
610  bool bREAD_nWRITE,
611  bool bCHECK_ONLY
612 )
613 {
614  /* Block Transfers Mode */

616

618  int32_t PNTR_offset;
619

622 
623  /* Lets just use 8bits Mode SPI transfers for patch*/
624  {
625  /* Setup pointer to Host start of data */
626  uint8_t * pHostMem = (uint8_t *)pBLOCK->pData;
627  uint8_t * pHostRef = NULL;
628 
629  if(bCHECK_ONLY)
630  {
631  pHostRef = pHostMem;
632  pHostMem = NULL;
633  }
634 
635  uint32_t size = pBLOCK->Size;
636 
637  /* No valid PHY SPI pointer available, use 8bits access with no offset */
638  if(PNTR_id == 0)

642  {
643  return ADI_ADF7030_1_SPI_COMM_FAILED;
644  }
645 
646  if(bREAD_nWRITE == true)
647  {
648  /* Read from radio PHY */

650  PNTR_CUSTOM0_ADDR,
651  size,
652  pHostMem,
653  pHostRef) != ADI_ADF7030_1_SUCCESS)
654  {
655  return ADI_ADF7030_1_SPI_COMM_FAILED;
656  }
657  }

665  {
666  return ADI_ADF7030_1_SPI_COMM_FAILED;
667  }
668  }
669  }

671  else
672  {
673  if(bREAD_nWRITE == true)
674  {
675  /* Read from radio PHY */
676  if(adf7030_1__SPI_rd_cmp_byte_b_a( pSPIDevInfo,
677  PNTR_id,
678  PNTR_offset,
679  size,
680  pHostMem,
