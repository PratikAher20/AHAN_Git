551 ADI_ADF7030_1_RESULT adf7030_1__XferCmdPoll(
552  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,
553  uint8_t nPhyCmd,
554  uint32_t nStatusPoll,
555  ADI_ADF7030_1_RADIO_STATE nStatePoll,
556  uint8_t nRetry
557 )
558 {
559  ADI_ADF7030_1_RESULT eReturn = ADI_ADF7030_1_SUCCESS;
560 
561  if(nPhyCmd)
562  {
563  /* Transmit the sequence */
564  adf7030_1__SPI_ReadWrite_Fast( pSPIDevInfo,
565  &nPhyCmd,
566  &pSPIDevInfo->nStatus.VALUE,
567  1 );
568  }
569 
570  /* Get the SPI status polling mask, assume*/
571  uint8_t nStatusPoll_Cond1 = *((uint8_t *)&nStatusPoll + 0);
572  uint8_t nStatusPoll_Cond2 = *((uint8_t *)&nStatusPoll + 1);
573  uint8_t nStatusPoll_msk = *((uint8_t *)&nStatusPoll + 2);
574 
575  nStatusPoll_Cond1 &= nStatusPoll_msk;
576  nStatusPoll_Cond2 &= nStatusPoll_msk;
577 
578  if((nStatePoll == 0) && (nStatusPoll_msk != 0))
579  {
580  /* Poll until SPI status match */
581  uint32_t nPollcnt = nRetry;
582 
583  uint8_t NOP_Cmd = 0xFF;
584 
585  bool EXIT = false;
586 
587  /* Polling loop */
588  do{
589  /* Decrement the polling loop counter if tracking */
590  if(nRetry) nPollcnt--;
591 
592  /* Transmit the sequence */
593  adf7030_1__SPI_ReadWrite_Fast( pSPIDevInfo,
594  &NOP_Cmd,
595  &pSPIDevInfo->nStatus.VALUE,
596  1 );
597 
598  /* Set Exit flag if SPI Status word matches nStatusPoll_Cond1 */
599  if(nStatusPoll_Cond1 == (pSPIDevInfo->nStatus.VALUE & nStatusPoll_msk)){
600  EXIT = true;
601  }
602 
603  /* Set Exit flag if SPI Status word matches nStatusPoll_Cond2 */
604  if(nStatusPoll_Cond2 == (pSPIDevInfo->nStatus.VALUE & nStatusPoll_msk)){
605  EXIT = true;
606  }
607 
608  /* Return if number of retries expired in the polling loop */
609  if(nRetry && !nPollcnt)
610  {
611  return ADI_ADF7030_1_SPI_DEV_POLL_EXPIRE;
612  }
613 
614  }while(EXIT == false);
615  }
616  else if(nStatePoll)
617  {
618  /* Poll until PHY state match */
619  uint32_t Addr = MISC_FW_Addr;
620 
621  /* Get current SPI custom pntr 2 value */
622  int32_t AddrDiff = (int32_t)Addr - (int32_t)(pSPIDevInfo->PHY_PNTR[PNTR_CUSTOM2_ADDR]);
623 
624  uint8_t Offset;
625 
626  if((AddrDiff >= 0 ) && (AddrDiff < (255 - 1)))
627  {
628  /* Here PNTR_CUSTOM2_ADDR is within range, just change the offset */
629  Offset = (uint8_t)AddrDiff;
630 
631  }else{
632  /* Here PNTR_CUSTOM2_ADDR is not within range, changing it, we shall */
633 
634  /* Setup the PHY Radio SPI pointer 0 address to Addr */
635  if(adf7030_1__SPI_wr_word_b_a( pSPIDevInfo, pSPIDevInfo->PHY_PNTR[PNTR_SETUP_ADDR] + 8, 1, &Addr) != ADI_ADF7030_1_SUCCESS)
636  {
637  return ADI_ADF7030_1_SPI_COMM_FAILED;
638  }
639 
640  pSPIDevInfo->PHY_PNTR[PNTR_CUSTOM2_ADDR] = Addr;
641 
642  Offset = 0;
643  }
644 
645  /* Setup SPI tx and rx buffer addresses */
646  uint8_t * pSPI_TX_BUFF = pSPIDevInfo->pSPI_TX_BUFF;
647  uint8_t * pSPI_RX_BUFF = pSPIDevInfo->pSPI_RX_BUFF;
648 
649  /* Transmit SPI command - use 1bytes offset to keep rest of data aligned on word boundary */
650  *(pSPI_TX_BUFF + 0) = ADF703x_SPI_MEM_READ |
651  ADF703x_SPI_MEM_BLOCK |
652  ADF703x_SPI_MEM_ADDRESS |
653  ADF703x_SPI_MEM_SHORT |
654  PNTR_CUSTOM2_ADDR;
655 
656  /* Set Block offset from pntrID */
657  *(pSPI_TX_BUFF + 1) = Offset + 1;
658 
659  /* Add 8bits of NOPs in spi_txbuf */
660  *((uint16_t *)pSPI_TX_BUFF + 1) = 0xFFFF;
661 
662  bool EXIT = false;
663 
664  uint32_t nPollcnt = nRetry;
665 
666  /* Polling loop */
667  do{
668  /* Transmit the sequence */
669  adf7030_1__SPI_ReadWrite_Fast( pSPIDevInfo,
670  (void *)pSPI_TX_BUFF,
671  (void *)pSPI_RX_BUFF,
672  4 );
673 
674  /* Write back current SPI Status into pSPIDevInfo structure */
675  pSPIDevInfo->nStatus.VALUE = *(pSPI_RX_BUFF + 2);
676 
677  /* Write back current SPI Status into pSPIDevInfo structure */
678  pSPIDevInfo->nPhyState = (ADI_ADF7030_1_RADIO_STATE)*(pSPI_RX_BUFF + 3);
679 
680  /* Decrement the polling loop counter if tracking */
681  if(nRetry) nPollcnt--;
682 
683  if(nStatusPoll_msk){
684  /* Set Exit flag if SPI Status word matches nStatusPoll_Cond1 and FW State matches nStatePoll */
685  if((nStatusPoll_Cond1 == (*(pSPI_RX_BUFF + 2) & nStatusPoll_msk)) && (nStatePoll == *(pSPI_RX_BUFF + 3))){
686  EXIT = true;
687  }
688  /* Set Exit flag if SPI Status word matches nStatusPoll_Cond2 and FW State matches nStatePoll */
689  if((nStatusPoll_Cond2 == (*(pSPI_RX_BUFF + 2) & nStatusPoll_msk)) && (nStatePoll == *(pSPI_RX_BUFF + 3))){
690  EXIT = true;
691  }
692  }else{
693  /* Set Exit flag if FW State matches nStatePoll */
694  if(nStatePoll == *(pSPI_RX_BUFF + 3)){
695  EXIT = true;
696  }
697  }
698 
699  /* Return if number of retries expired in the polling loop */
700  if(nRetry && !nPollcnt)
701  {
702  return ADI_ADF7030_1_SPI_DEV_POLL_EXPIRE;
703  }
704  }while(EXIT == false);
705  }
706
