376 
377  /* Pointer to the ADF7030-1 instance information */
378  ADI_ADF7030_1_INFO *pDevInfo = pDevice->pDevInfo;
379

382 
383  /* ADF7030-1 result code */
384  ADI_ADF7030_1_RESULT eResult;
385 
386 #ifdef ADI_DEBUG
387  /* Validate the given device handle */
388  if((eResult = ValidateHandle(hDevice)) != ADI_ADF7030_1_SUCCESS)
389  {
390  return eResult;
391  }
392 #endif /* ADI_DEBUG */
393 
394 #if (ADF7030_1_PHY_ERROR_REPORT_ENABLE == 1)
395  /* Disable automatic PhyError reporting by default */
396  pSPIDevInfo->bPhyErrorCheck = false;
397 #endif
398 
399  if(bEnable == true)
400  {
401  /* Place holder for turning on the adf7030-1 PHY */
402 
403  /* Execute user defined "enable" sequence */
404  if(pDevInfo->pfEnableFunc)
405  {
406  if((eResult = pDevInfo->pfEnableFunc(pDevInfo)) != ADI_ADF7030_1_SUCCESS)
407  {
408  return eResult;
409  }
410  }
411  else
412  {
413  /* Execute default "enable" sequence */
414 
415  /* Toggle the Reset pin the the PHY Radio */
416  if((pDevInfo->ResetGPIOInfo.ePort != 0UL) && (pDevInfo->ResetGPIOInfo.nPin != 0UL))
417  {

422  }
423

425  if((eResult = adf7030_1__STATE_PhyCMD_WaitReady( pSPIDevInfo, PHY_OFF, PHY_OFF )) != ADI_ADF7030_1_SUCCESS)
426  {
427  return eResult;
428  }
429 
430  /* Send the Default usecase configuration to the part if specified */
431  if((pDevInfo->pStartupCfg != NULL) && (pDevInfo->StartupCfgSize != 0UL))
432  {
433  if((eResult = adf7030_1__SendConfiguration( pSPIDevInfo,
434  (uint8_t *)pDevInfo->pStartupCfg,
435  pDevInfo->StartupCfgSize )) != ADI_ADF7030_1_SUCCESS)
436  {
437  return eResult;
438  }
439  }
440  }
441 
442  pDevice->eState |= ADI_ADF7030_1_STATE_ENABLED;
443  }
444  else
445  {
446  /* Place holder for turning off the adf7030-1 PHY */
447 
448  /* Clear interrupt enable register for IRQ0 */
449  if(adf7030_1__IRQ_ClrAllStatus(pDevInfo, ADI_ADF7030_1_INTPIN0) != ADI_ADF7030_1_SUCCESS)
450  {
451  return ADI_ADF7030_1_SPI_DEV_FAILED;
452  }
453 
454  /* Clear interrupt enable register for IRQ1 */
455  if(adf7030_1__IRQ_ClrAllStatus(pDevInfo, ADI_ADF7030_1_INTPIN1) != ADI_ADF7030_1_SUCCESS)
456  {
457  return ADI_ADF7030_1_SPI_DEV_FAILED;
458  }
459 
460  /* Execute user defined "disable" sequence */
461  if(pDevInfo->pfDisableFunc)
462  {
463  if((eResult = pDevInfo->pfDisableFunc(pDevInfo)) != ADI_ADF7030_1_SUCCESS)
464  {
465  return eResult;
466  }
467  }
468  else
469  {
470  /* Reset the PHY Radio on exit */
471  if((eResult = adf7030_1__STATE_PhyShutdown(pSPIDevInfo)) != ADI_ADF7030_1_SUCCESS)
472  {
473  return eResult;
474  }
475  }
476 
477  /* Change driver state */
478  pDevice->eState &= ~ADI_ADF7030_1_STATE_ENABLED;
479  }
480 
481  return ADI_ADF7030_1_SUCCESS;
482 }

519 ADI_ADF7030_1_RESULT adf7030_1__SetupHostPhyCfg(
520  ADI_ADF7030_1_HANDLE const hDevice,
521  void * const pCfgMem,
522  uint32_t nCfgMemSize,
523  MEM_CFG_DESCRIPTOR_Type * pCfgRef,
524  bool bInit
525 )
526 {
527  /* Pointer to the ADF7030-1 device */
528  ADI_ADF7030_1_DEVICE *pDevice = (ADI_ADF7030_1_DEVICE *)hDevice;
529 
530  /* Pointer to the ADF7030-1 instance information */
531  ADI_ADF7030_1_INFO *pDevInfo = pDevice->pDevInfo;
532 
533  /* Pointer to the ADF7030-1 instance SPI information */
534  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo = &pDevInfo->SPIInfo;
535 
536  /* Size of structure is equal to pCfgRef->Size (total size of data contained in all blocks)
537  + 28bytes (header size)
538  + ((sizeof(DATA_BLOCK_DESCRIPTOR_Type) + 4) * (16 blocks max)) ... 4 is to allow word alligned block start
539  */
540  if(nCfgMemSize <= (pCfgRef->Size + 28 + ( 16 * (sizeof(DATA_BLOCK_DESCRIPTOR_Type) + 4))))
541  {
542  return ADI_ADF7030_1_INSUFFICIENT_MEMORY;
543  }
544 
545  /* Initialize header of the configuration */
546  memcpy(pCfgMem, (const void *)pCfgRef, 28);
547 
548  /* Declare pointer to the new configuration structure */
549  MEM_CFG_DESCRIPTOR_Type * pCfg = (MEM_CFG_DESCRIPTOR_Type *)pCfgMem;
550 
551  /* Remove pointer to info and nested structure if any */
552  pCfg->pInfo = NULL;
553  pCfg->pNext = NULL;
554 
555  /* Set the start address for first BLOCK.pData */
556  uint32_t BlocAddr = (uint32_t)&pCfg->BLOCKS + (16 * sizeof(DATA_BLOCK_DESCRIPTOR_Type));
557 
558  /* Initialise each cfg block headers */
559  for(uint32_t i = 0; i < pCfg->nbBlock; i++)
560  {
561  /* Copy each block description */
562  memcpy((void *)&pCfg->BLOCKS[i], (const void *)&pCfgRef->BLOCKS[i], sizeof(DATA_BLOCK_DESCRIPTOR_Type));
563 
564  /* Change block data pointers to new ram location */
565  pCfg->BLOCKS[i].pData = (uint8_t *)BlocAddr;
566 
567  /* Clear each block data buffer */
568  memset((void *)BlocAddr, 0, pCfg->BLOCKS[i].Size);
569 
570  BlocAddr += pCfg->BLOCKS[i].Size;
571 
572  /* Realign each pBloc on word boundaries */
573  if((BlocAddr & 0x3ul) != 0)
574  {
575  BlocAddr = (((BlocAddr >> 2) + 1) << 2);
576  }
577  }
578 
579  /* Setup instance pointer to configuration structure */
580  pDevInfo->pCfg = pCfg;
581 
582  /* Setup instance pointer to Host Shadow "Profile" structure */
583  pDevInfo->pPROFILE__BLOCK = &pCfg->BLOCKS[9];
584 
585  /* Setup instance pointer to Host Shadow "Packet" structure */
586  pDevInfo->pPACKET_CFG__BLOCK = &pCfg->BLOCKS[10];
587 
588  /* Setup instance pointer to Host Shadow "TX packet" buffer */
589  pDevInfo->pTX_PCKBUFF__BLOCK = &pCfg->BLOCKS[14];
590 
591  /* Setup instance pointer to Host Shadow "RX packet" buffer */
592  pDevInfo->pRX_PCKBUFF__BLOCK = &pCfg->BLOCKS[15];
593 
594  if(bInit == true)
595  {
596  /* Checking correct shadow profile size */
597  if(pDevInfo->pPROFILE__BLOCK->Size != sizeof(ADI_PROFILE_Type))
598  {
599  return ADI_ADF7030_1_INVALID_PHY_CONFIGURATION;
600  }
601 
602  /* Reading PHY Radio profile location */
603  uint32_t Profile_Addr;
604  if(adf7030_1__Get_PhyProfileAddr(pDevInfo, &Profile_Addr ) != ADI_ADF7030_1_SUCCESS)
605  {
606  return ADI_ADF7030_1_INVALID_PHY_CONFIGURATION;
607  }
608 
609  /* Override PHY profile address with default value */
610  pDevInfo->pPROFILE__BLOCK->Addr = Profile_Addr;
611 
612  /* Readback Profile data from PHY Radio */
613  if(adf7030_1__ReadDataBlock(pSPIDevInfo, pDevInfo->pPROFILE__BLOCK) != ADI_ADF7030_1_SUCCESS)
614  {
615  return ADI_ADF7030_1_SPI_COMM_FAILED;
616  }
617 
618  /* Checking correct shadow packet config address and size*/
619  if(pDevInfo->pPACKET_CFG__BLOCK->Size != sizeof(ADI_GENERIC_PKT_Type))
620  {
621  return ADI_ADF7030_1_INVALID_PHY_CONFIGURATION;
622  }
623 
624  /* Reading PHY Radio packet config location */
625  uint32_t PacketConfig_Addr;
626  if(adf7030_1__Get_PhyPckConfigAddr(pDevInfo, &PacketConfig_Addr ) != ADI_ADF7030_1_SUCCESS)
627  {
628  return ADI_ADF7030_1_INVALID_PHY_CONFIGURATION;
629  }
630 
631  /* Override PHY packet config address with default value */
632  pDevInfo->pPACKET_CFG__BLOCK->Addr = PacketConfig_Addr;
633 
634  /* Readback Packet configuration data from PHY Radio */
635  if(adf7030_1__ReadDataBlock(pSPIDevInfo, pDevInfo->pPACKET_CFG__BLOCK) != ADI_ADF7030_1_SUCCESS)
636  {
637  return ADI_ADF7030_1_SPI_COMM_FAILED;
638  }
639  }
640 
641  return ADI_ADF7030_1_SUCCESS;
642 }

664 ADI_ADF7030_1_RESULT adf7030_1__SetupHostTrxBuffer(
665  ADI_ADF7030_1_INFO * pDevInfo,
666  void * const pTrxMem,
667  uint32_t nTrxMemSize
668 )
669 {
670  if(pDevInfo->pPACKET_CFG__BLOCK != NULL)
671  {
672  /* Pointer to Host Shadow Packet Configuration structure */
673  ADI_GENERIC_PKT_Type * pGENERIC_PKT = (ADI_GENERIC_PKT_Type *)pDevInfo->pPACKET_CFG__BLOCK->pData;
674 
675  uint16_t ptr_tx_base = pGENERIC_PKT->BUFF_CFG0_b.PTR_TX_BASE;
676  uint16_t ptr_rx_base = pGENERIC_PKT->BUFF_CFG0_b.PTR_RX_BASE;
677  uint32_t PHY_Base_Addr = 0x20000000UL;
678 
679 #ifdef __INTERNAL_DEV__
680  // Get the TX buffer memory location and payload size
681  if(pGENERIC_PKT->BUFF_CFG0_b.EXTENDED_BUFF_EN)
682  {
683 
684  //Assign pTX_PCKBUFF__BLOCK to Extended shadow packet buffer
685  pDevInfo->pTX_PCKBUFF__BLOCK->Size = pGENERIC_PKT->BUFF_CFG1_BIS_b.TRX_SIZE;
686  pDevInfo->pTX_PCKBUFF__BLOCK->Addr = PHY_Base_Addr | (((pGENERIC_PKT->BUFF_CFG0_b.EXTENDED_BUFF_BASE_SEL) ? ptr_tx_base : ptr_rx_base) << 2 );
687  pDevInfo->pTX_PCKBUFF__BLOCK->pData = (uint8_t *)pTrxMem;
688 
689  if(pDevInfo->pTX_PCKBUFF__BLOCK->Size > nTrxMemSize)
690  {
691  return ADI_ADF7030_1_INSUFFICIENT_MEMORY;
692  }
693 
694  /* Clear Host TX payload buffer */
695  memset((void *)pDevInfo->pTX_PCKBUFF__BLOCK->pData, 0, pDevInfo->pTX_PCKBUFF__BLOCK->Size);
696  }
697  else
698 #endif
699  {
700  //Assign pTX_PCKBUFF__BLOCK to TX shadow packet buffer at
701  pDevInfo->pTX_PCKBUFF__BLOCK->Size = pGENERIC_PKT->BUFF_CFG1_b.TX_SIZE;
702  pDevInfo->pTX_PCKBUFF__BLOCK->Addr = PHY_Base_Addr | (ptr_tx_base << 2);
703 
704  //Setup pData @ the midle of reserved Host memory
705  pDevInfo->pTX_PCKBUFF__BLOCK->pData = (uint8_t *)pTrxMem;
706 
707  if(pDevInfo->pTX_PCKBUFF__BLOCK->Size > (nTrxMemSize >> 1))
708  {
709  return ADI_ADF7030_1_INSUFFICIENT_MEMORY;
710  }
711 
712  /* Clear Host RX payload buffer */
713  memset((void *)pDevInfo->pTX_PCKBUFF__BLOCK->pData, 0, pDevInfo->pTX_PCKBUFF__BLOCK->Size);
714 
715  //Assign pRX_PCKBUFF__BLOCK to RX shadow packet buffer
716  pDevInfo->pRX_PCKBUFF__BLOCK->Size = pGENERIC_PKT->BUFF_CFG1_b.RX_SIZE;
717  pDevInfo->pRX_PCKBUFF__BLOCK->Addr = PHY_Base_Addr | (ptr_rx_base << 2);
718 
719  //Setup pData @ the start of reserved Host memory
720  pDevInfo->pRX_PCKBUFF__BLOCK->pData = (uint8_t *)((uint32_t)pTrxMem + (nTrxMemSize >> 1));
721 
722  if(pDevInfo->pRX_PCKBUFF__BLOCK->Size > (nTrxMemSize >> 1))
723  {
724  return ADI_ADF7030_1_INSUFFICIENT_MEMORY;
725  }
726 
727  /* Clear Host RX payload buffer */
728  memset((void *)pDevInfo->pRX_PCKBUFF__BLOCK->pData, 0, pDevInfo->pRX_PCKBUFF__BLOCK->Size);
729  }
730  }
731  else
732  {
733  /* Operation not supported without host shadow memory */
734  return ADI_ADF7030_1_INVALID_PHY_CONFIGURATION;
735  }
736 
737  return ADI_ADF7030_1_SUCCESS;
738 }
