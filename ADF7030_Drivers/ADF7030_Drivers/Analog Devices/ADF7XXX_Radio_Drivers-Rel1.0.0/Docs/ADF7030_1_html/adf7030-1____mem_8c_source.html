201 )
202 {
203  /* ADF7030-1 result code */
204  ADI_ADF7030_1_RESULT eResult = ADI_ADF7030_1_SPI_COMM_FAILED;
205 
206  bool bEXIT;
207  do{
208 
209  for(uint32_t i = 0; i < pCONFIG->nbBlock; i++)
210  {
211  if((eResult = adf7030_1__XferDataBlock( pSPIDevInfo,

217  {
218  return eResult;

220  }
221

224  {

227  bEXIT = false;
228  }
229  else
230  {
231  bEXIT = true;

234  }while(bEXIT == false);

460 ADI_ADF7030_1_RESULT adf7030_1__XferDataBlock(
461  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,
462  DATA_BLOCK_DESCRIPTOR_Type *pBLOCK,
463  uint16_t Byte_Offset,
464  uint16_t Byte_Size,
465  bool bREAD_nWRITE,
466  bool bCHECK_ONLY
467 )
468 {
469  /* 8bits Mode SPI transfers mode by default */
470  if(pBLOCK->WordXfer == 0)
471  {
472  /* Get Memory Block Address */
473  uint32_t PHYAddr = pBLOCK->Addr + Byte_Offset;
474 
475  uint32_t size = (Byte_Size == 0) ? pBLOCK->Size : Byte_Size;
476 
477  ADI_ADF7030_1_RADIO_SPI_PNTR_TYPE PNTR_id;
478  int32_t PNTR_offset;
479 
480  /* Get good PNTRid from pSPIDevInfo->PHY_PNTR[] */
481  adf7030_1__SPI_FindMMapPointer( pSPIDevInfo, PHYAddr, &PNTR_id, &PNTR_offset);
482 
483  /* Setup pointer to Host start of data */
484  uint8_t * pHostMem = (uint8_t *)(pBLOCK->pData + Byte_Offset);
485  uint8_t * pHostRef = NULL;
486 
487  if(bCHECK_ONLY)
488  {
489  pHostRef = pHostMem;
490  pHostMem = NULL;
491  }
492 
493  /* No valid PHY SPI pointer available, use 8bits access with no offset */
494  if(PNTR_id == 0)
495  {
496  /* Change the SPI custom pointer 0 */
497  if(adf7030_1__SPI_SetMMapCustomPntr0( pSPIDevInfo, PHYAddr ) != ADI_ADF7030_1_SUCCESS)
498  {
499  return ADI_ADF7030_1_SPI_COMM_FAILED;
500  }
501 
502  if(bREAD_nWRITE == true)
503  {
504  /* Read from radio PHY */
505  if(adf7030_1__SPI_rd_cmp_byte_p_a( pSPIDevInfo,
506  PNTR_CUSTOM0_ADDR,
507  size,
508  pHostMem,
509  pHostRef) != ADI_ADF7030_1_SUCCESS)
510  {
511  if(pBLOCK->Volatile)
512  {
513  /* In case block contains volatile data, force return SUCCESS for debug */
514  //\todo Return different value
515  return ADI_ADF7030_1_SUCCESS;
516  }
517  else
518  {
519  return ADI_ADF7030_1_SPI_COMM_FAILED;
520  }
521  }
522  }else{
523  /* Write to radio PHY */
524  if(adf7030_1__SPI_wr_byte_p_a( pSPIDevInfo, PNTR_CUSTOM0_ADDR, size, pHostMem) != ADI_ADF7030_1_SUCCESS)
525  {
526  return ADI_ADF7030_1_SPI_COMM_FAILED;
527  }
528  }
529  }
530  /* Valid PHY SPI pointer available, use 8bits access with pointer offset */
531  else
532  {
533  if(bREAD_nWRITE == true)
534  {
535  /* Read from radio PHY */
536  if(adf7030_1__SPI_rd_cmp_byte_b_a( pSPIDevInfo,
537  PNTR_id,
538  PNTR_offset,
539  size,
540  pHostMem,
541  pHostRef) != ADI_ADF7030_1_SUCCESS)
542  {
543  if(pBLOCK->Volatile)
544  {
545  /* In case block contains volatile data, force return SUCCESS for debug */
546  //\todo Return different value
547  return ADI_ADF7030_1_SUCCESS;
548  }
549  else
550  {
551  return ADI_ADF7030_1_SPI_COMM_FAILED;
552  }
553  }
554  }else{
555  /* Write to radio PHY */
556  if(adf7030_1__SPI_wr_byte_b_a( pSPIDevInfo, PNTR_id, PNTR_offset, size, pHostMem) != ADI_ADF7030_1_SUCCESS)
557  {
558  return ADI_ADF7030_1_SPI_COMM_FAILED;
559  }
560  }
561  }
562  }
563  /* This block only support 32bits Mode SPI transfers, Let us just use block modes */
564  else
565  {
566  /* Intercept non-word transfer error due to Byte_Offset */
567  if(Byte_Offset & 0x3)
568  {
569  return ADI_ADF7030_1_INVALID_OPERATION;
570  }
571 
572  /* Get Memory Block Address */
573  uint32_t PHYAddr = pBLOCK->Addr + Byte_Offset;
574 
575  uint32_t size = (Byte_Size == 0) ? (pBLOCK->Size >> 2) : (Byte_Size >> 2);
576 
577  /* Setup pointer to Host start of data */
578  uint32_t * pHostMem = (uint32_t *)(pBLOCK->pData + Byte_Offset);
579  uint32_t * pHostRef = NULL;
580 
581  if(bCHECK_ONLY)
582  {
583  pHostRef = pHostMem;
584  pHostMem = NULL;
585  }
586 
587  if(bREAD_nWRITE == true)
588  {
589  /* Read from radio PHY */
590  if(adf7030_1__SPI_rd_cmp_word_b_a( pSPIDevInfo,
591  PHYAddr,
592  size,
593  pHostMem,
594  pHostRef) != ADI_ADF7030_1_SUCCESS)
595  {
596  if(pBLOCK->Volatile)
597  {
598  /* In case block contains volatile data, force return SUCCESS for debug */
599  //\todo Return different value
600  return ADI_ADF7030_1_SUCCESS;
601  }
602  else
603  {
604  return ADI_ADF7030_1_SPI_COMM_FAILED;
605  }
606  }
607  }else{
608  /* Write to radio PHY */
609  if(adf7030_1__SPI_wr_word_b_a( pSPIDevInfo, PHYAddr, size, pHostMem) != ADI_ADF7030_1_SUCCESS)
610  {
611  return ADI_ADF7030_1_SPI_COMM_FAILED;
612  }
613  }
614  }
615 
616  return ADI_ADF7030_1_SUCCESS;
617 }

641 ADI_ADF7030_1_RESULT adf7030_1__SPI_Xfer_Mem(
642  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,
643  MEM_DESCRIPTOR_Type * pMEM,
644  bool bREAD_nWRITE
645 )
646 {
647  bool bEXIT;
648  do{
649 
650  if(!pMEM->XferMode)
651  {
652  /* Block Transfers Mode */
653 
654  uint32_t PHYAddr = pMEM->PhyAddr + pMEM->BlockXfer.Offset;
655 
656  ADI_ADF7030_1_RADIO_SPI_PNTR_TYPE PNTR_id;
657  int32_t PNTR_offset;
658 
659  /* Get good PNTRid from pSPIDevInfo->PHY_PNTR[] */
660  adf7030_1__SPI_FindMMapPointer( pSPIDevInfo, PHYAddr, &PNTR_id, &PNTR_offset);
661 
662  /* 32bits Mode SPI transfers */
663  if(!(PHYAddr & 3) && !(pMEM->BlockXfer.Size & 3))
664  {
665  /* Setup pointer to Host start of data */
666  uint32_t * pHostMem;
667  pHostMem = (uint32_t *)pMEM->pBuff + (pMEM->BlockXfer.Offset >> 2);
668 
669  /* Valid PHY SPI pointer available, use 32bits bloc pointer access */
670  if((PNTR_id) && !(PNTR_offset >> 2))
671  {
672  if(bREAD_nWRITE == true)
673  {
674  /* Read from radio PHY */
675  if(adf7030_1__SPI_rd_word_b_p( pSPIDevInfo, PNTR_id, pMEM->BlockXfer.Size >> 2, pHostMem) != ADI_ADF7030_1_SUCCESS)
676  {
677  return ADI_ADF7030_1_SPI_COMM_FAILED;
678  }
679  }else{
680  /* Write to radio PHY */
681  if(adf7030_1__SPI_wr_word_b_p( pSPIDevInfo, PNTR_id, pMEM->BlockXfer.Size >> 2, pHostMem) != ADI_ADF7030_1_SUCCESS)
682  {
683  return ADI_ADF7030_1_SPI_COMM_FAILED;
684  }
685  }
686  }
687  /* No valid PHY SPI pointer available, use 32bits bloc address access */
688  else
689  {
690  if(bREAD_nWRITE == true)
691  {
692  /* Read from radio PHY */
693  if(adf7030_1__SPI_rd_word_b_a( pSPIDevInfo, PHYAddr, pMEM->BlockXfer.Size >> 2, pHostMem) != ADI_ADF7030_1_SUCCESS)
694  {
695  return ADI_ADF7030_1_SPI_COMM_FAILED;
696  }
697  }else{
698  /* Write to radio PHY */
699  if(adf7030_1__SPI_wr_word_b_a( pSPIDevInfo, PHYAddr, pMEM->BlockXfer.Size >> 2, pHostMem) != ADI_ADF7030_1_SUCCESS)
700  {
701  return ADI_ADF7030_1_SPI_COMM_FAILED;
702  }
703  }
704  }
705  }
706  /* 8bits Mode SPI transfers */
707  else
708  {
709  /* Setup pointer to Host start of data */
710  uint8_t * pHostMem;
711  pHostMem = (uint8_t *)pMEM->pBuff + pMEM->BlockXfer.Offset;
712 
713  /* No valid PHY SPI pointer available, use 8bits access with no offset */
714  if(PNTR_id == 0)
715  {
716  /* Change the SPI custom pointer 0 */
717  if(adf7030_1__SPI_SetMMapCustomPntr0( pSPIDevInfo, PHYAddr ) != ADI_ADF7030_1_SUCCESS)
718  {
719  return ADI_ADF7030_1_SPI_COMM_FAILED;
720  }
721 
722  /* Copy Host MEM memory block to PHY memory block via the SPI */
723  if(bREAD_nWRITE == true)
724  {
725  /* Read from radio PHY */
726  if(adf7030_1__SPI_rd_byte_p_a( pSPIDevInfo, PNTR_CUSTOM0_ADDR, pMEM->BlockXfer.Size, pHostMem) != ADI_ADF7030_1_SUCCESS)
727  {
728  return ADI_ADF7030_1_SPI_COMM_FAILED;
729  }
730  }else{
731  /* Write to radio PHY */
732  if(adf7030_1__SPI_wr_byte_p_a( pSPIDevInfo, PNTR_CUSTOM0_ADDR, pMEM->BlockXfer.Size, pHostMem) != ADI_ADF7030_1_SUCCESS)
733  {
734  return ADI_ADF7030_1_SPI_COMM_FAILED;
735  }
736  }
737  }
738  /* Valid PHY SPI pointer available, use 8bits access with pointer offset */
739  else
740  {
741  /* Copy Host MEM memory block to PHY memory block via the SPI */
742  if(bREAD_nWRITE == true)
743  {
744  /* Read from radio PHY */
745  if(adf7030_1__SPI_rd_byte_b_a( pSPIDevInfo, PNTR_id, PNTR_offset, pMEM->BlockXfer.Size, pHostMem) != ADI_ADF7030_1_SUCCESS)
746  {
747  return ADI_ADF7030_1_SPI_COMM_FAILED;
748  }
749  }else{
750  /* Write to radio PHY */
751  if(adf7030_1__SPI_wr_byte_b_a( pSPIDevInfo, PNTR_id, PNTR_offset, pMEM->BlockXfer.Size, pHostMem) != ADI_ADF7030_1_SUCCESS)
752  {
753  return ADI_ADF7030_1_SPI_COMM_FAILED;
754  }
755  }
756  }
757  }
758  }
759  else
760  {
761  /* Random Transfers Mode */
762 
763  }
764 
765 
766  /* Check if nested MEM transfert is present */
767  if(pMEM->pNextBloc)
768  {
769  /* Move pointer to MEM structure to the following one */
770  pMEM = pMEM->pNextBloc;
771  bEXIT = false;
772  }
773  else
774  {
775  bEXIT = true;
776  }
777 
778  }while(bEXIT == false);
779 
780  return ADI_ADF7030_1_SUCCESS;
781 }

1009 void adf7030_1__SPI_SetBytes(
1010  ADI_ADF7030_1_SPI_INFO * pSPIDevInfo,
1011  uint32_t Addr,
1012  uint32_t Value,
1013  uint32_t nbBytes,
1014  uint32_t * pRegVal
1015 )
1016 {
1017  /* SPI transfert result */
1018  ADI_ADF7030_1_RESULT eResult = ADI_ADF7030_1_SUCCESS;
1019 
1020  /* Get current SPI custom pntr 0 value */
1021  int32_t AddrDiff = (int32_t)Addr - (int32_t)(pSPIDevInfo->PHY_PNTR[PNTR_CUSTOM0_ADDR]);
1022 
1023  uint8_t Offset;
1024 
1025  /* Check if byte access is allowed at this address */
1026  bool bByteAccess = adf7030_1__MEM_CheckByteAccess(Addr);
1027  bool bInrange = ((AddrDiff >= 0) & (AddrDiff < (255 + nbBytes))) ? true : false;
1028 
1029  if((bInrange == true) && (bByteAccess == true))
1030  {
1031  /* Here PNTR_CUSTOM0_ADDR is within range, just change the offset */
1032  Offset = (uint8_t)AddrDiff;
1033  }else{
1034  /* Here PNTR_CUSTOM0_ADDR is not within range, changing it, we shall */
1035 
1036  if(((Addr >> 2) << 2) != pSPIDevInfo->PHY_PNTR[PNTR_CUSTOM0_ADDR]) //Fix for double setup of pntr0
1037  {
1038  /* Setup the PHY Radio SPI pointer 0 address to Addr */
1039  eResult = adf7030_1__SPI_SetMMapCustomPntr0( pSPIDevInfo, ((Addr >> 2) << 2));
1040 
1041  /* Exit if error */
1042  if(eResult != ADI_ADF7030_1_SUCCESS)
1043  {
1044  pSPIDevInfo->eXferResult = eResult;
1045  return;
1046  }
1047  }
1048 
1049  Offset = Addr & 0x03;
1050  }
1051 
1052  if(bByteAccess == false)
1053  {
1054  /* Construct the byte mask depending on nbBytes */
1055  uint32_t msk = (0x80 << ((nbBytes - 1) << 3));
1056  msk |= ~msk;
1057  msk <<= (Offset << 3);
1058 
1059  uint32_t tmp_reg = 0;
1060 
1061  /* Clear masked bytes from pRegVal */
1062  if(pRegVal)
1063  {
1064  tmp_reg = *pRegVal & ~msk;
1065  }
1066 
1067  /* Shift Value by the Offset */
1068  Value <<= (Offset << 3);
1069 
1070  /* Mix value in full reg */
1071  Value |= tmp_reg;
1072 
1073  /* Byte access is not permitted, do word access instead */
1074  eResult = adf7030_1__SPI_wr_word_b_p( pSPIDevInfo, PNTR_CUSTOM0_ADDR, 1, &Value);
1075 
1076  }
1077  else
1078  {
1079  if((Offset == 0) && (nbBytes == 4))
1080  {
1081  /* Write the 32bit alligned data */
1082  eResult = adf7030_1__SPI_wr_word_b_p( pSPIDevInfo, PNTR_CUSTOM0_ADDR, 1, &Value);
1083  }else{
1084  /* Write the nbBytes data */
1085  eResult = adf7030_1__SPI_wr_byte_b_a( pSPIDevInfo, PNTR_CUSTOM0_ADDR, Offset, nbBytes, (uint8_t *)&Value);
1086  }
1087  }
1088 
1089  /* Exit if error */
1090  pSPIDevInfo->eXferResult = eResult;
